# Websocket socket.io events

- start = start music from the first song of the playlist
- stop = stop music after the song ends
- hard_stop = stop music in the middle of the song
- state = get the state of the admin panel
- status_update = for the frontend player

- get_playlist = for frontend request
- next = for skipping music
- restart = for restarting playlist
- auto = set to automatic playlist
- set_playlist for the frontend admin playlist settings
- get_state = for the admin panel info
- crossfade_enable = enable crossfade
- crossfade_set = for changing crossfade settings
- crossfade_preset = for changing crossfade by preset
